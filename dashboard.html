Restaurant Inspection Grades - Scatter Plot

nyc_inspections %>% 
  mutate(
    text_label = str_c("Restaurant: ", dba, 
                       "\nBorough: ", boro,
                       "\nViolation: ", violation_description)
  ) %>% 
  drop_na(grade) %>% 
  plot_ly( 
    x = ~inspection_date, 
    y = ~score, 
    color = ~grade, 
    colors = "viridis",
    type = "scatter", 
    mode = "markers",
    alpha = .4, 
    text = ~text_label
  )

Number of Critical Flags by Borough - Bar Chart

nyc_inspections %>% 
  group_by(boro) %>% 
  filter(boro != "Missing") %>% 
  count(critical_flag) %>% 
  ungroup() %>% 
  mutate(boro = fct_reorder(boro, -n),
         critical_flag = fct_reorder(critical_flag, -n)) %>% 
  plot_ly(
    x = ~boro, 
    y = ~n,
    color = ~critical_flag,
    colors = "viridis",
    type = "bar")
